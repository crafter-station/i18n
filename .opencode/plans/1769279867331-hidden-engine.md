# Real-Time Voice Translation: Low-Latency Architecture Plan

## Summary

| Decision | Choice |
|----------|--------|
| **STT Provider** | ElevenLabs Scribe (WebSocket) |
| **Translation** | Groq Llama 3.1 8B Instant (FREE, ~50ms) |
| **TTS Provider** | ElevenLabs (WebSocket streaming) |
| **Daily.co Transcription** | Remove completely |
| **Original Audio** | Mute completely |
| **Target Latency** | **~300-425ms** end-to-end |

## Overview

Transform the video conferencing app to support real-time voice translation with **~300-425ms end-to-end latency**. Users select a language and hear all other participants in that language.

**Key Changes**:
1. Use **ElevenLabs for both STT and TTS** via WebSocket (single vendor)
2. Use **Groq Llama 3.1 8B** for translation (fastest, free tier)
3. Remove Daily.co's built-in transcription entirely

## Current State (after recent commits)

The app currently uses:
- **Daily.co built-in transcription** (Deepgram nova-2) via `daily.startTranscription()`
- **ElevenLabs TTS** via Vercel AI SDK (batch, not streaming)
- **GPT-4o-mini** for translation (batch)

## Why ElevenLabs for Both?

| Aspect | Current (Daily+Deepgram+ElevenLabs) | Proposed (ElevenLabs Only) |
|--------|-------------------------------------|----------------------------|
| Vendors | 3 (Daily, Deepgram, ElevenLabs) | 2 (Daily, ElevenLabs) |
| STT Latency | ~300-500ms (Daily's pipeline) | ~200ms (direct WebSocket) |
| TTS Latency | ~400ms (batch HTTP) | ~75ms TTFB (WebSocket streaming) |
| Complexity | Server-side transcription | Client-side WebSocket |
| Control | Limited (Daily controls STT) | Full (we control the pipeline) |

```typescript
  commit: false,

```

```typescript
// Query parameters
```

## Implementation Plan

### Phase 1: ElevenLabs STT Hook

**File: `src/hooks/use-elevenlabs-stt.ts`**

```typescript
"use client";

import { useRef, useCallback, useState } from "react";

interface UseElevenLabsSTTOptions {
  apiKey: string;
  languageCode?: string;
  onPartial?: (text: string) => void;
  onCommitted?: (text: string) => void;
}

export function useElevenLabsSTT({
  apiKey,
  languageCode = "en",
  onPartial,
  onCommitted,
}: UseElevenLabsSTTOptions) {
  const wsRef = useRef<WebSocket | null>(null);
  const audioContextRef = useRef<AudioContext | null>(null);
  const processorRef = useRef<ScriptProcessorNode | null>(null);
  const [isListening, setIsListening] = useState(false);

  const startListening = useCallback(async (stream: MediaStream) => {
    // Connect to ElevenLabs Scribe WebSocket
    const wsUrl = new URL("wss://api.elevenlabs.io/v1/speech-to-text/realtime");
    wsUrl.searchParams.set("model_id", "scribe_v1");
    wsUrl.searchParams.set("language_code", languageCode);
    wsUrl.searchParams.set("commit_strategy", "vad");
    wsUrl.searchParams.set("vad_silence_threshold_secs", "1.0");
    wsUrl.searchParams.set("audio_format", "pcm_16000");

    const ws = new WebSocket(wsUrl.toString(), {
      headers: { "xi-api-key": apiKey },
    });
    
    // Note: Browser WebSocket doesn't support headers, use token instead
    // Get single-use token from /api/elevenlabs-token endpoint
    
    wsRef.current = ws;

    ws.onopen = () => {
      setIsListening(true);
    };

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      
      if (data.message_type === "partial_transcript") {
        onPartial?.(data.text);
      } else if (data.message_type === "committed_transcript") {
        onCommitted?.(data.text);
      }
    };

    // Set up audio processing
    audioContextRef.current = new AudioContext({ sampleRate: 16000 });
    const source = audioContextRef.current.createMediaStreamSource(stream);
    
    // Use ScriptProcessor to get raw PCM (AudioWorklet preferred for production)
    const processor = audioContextRef.current.createScriptProcessor(4096, 1, 1);
    processorRef.current = processor;

    processor.onaudioprocess = (e) => {
      if (ws.readyState !== WebSocket.OPEN) return;
      
      const inputData = e.inputBuffer.getChannelData(0);
      
      // Convert Float32 to Int16
      const int16Data = new Int16Array(inputData.length);
      for (let i = 0; i < inputData.length; i++) {
        int16Data[i] = Math.max(-32768, Math.min(32767, inputData[i] * 32768));
      }
      
      // Send as base64
      const base64 = btoa(String.fromCharCode(...new Uint8Array(int16Data.buffer)));
      
      ws.send(JSON.stringify({
        message_type: "input_audio_chunk",
        audio_base_64: base64,
        sample_rate: 16000,
        commit: false,
      }));
    };

    source.connect(processor);
    processor.connect(audioContextRef.current.destination);
  }, [apiKey, languageCode, onPartial, onCommitted]);

  const stopListening = useCallback(() => {
    processorRef.current?.disconnect();
    audioContextRef.current?.close();
    wsRef.current?.close();
    setIsListening(false);
  }, []);

  return { isListening, startListening, stopListening };
}
```

### Phase 2: ElevenLabs TTS WebSocket Hook

**File: `src/hooks/use-elevenlabs-tts.ts`**

```typescript
"use client";

import { useRef, useCallback } from "react";
import { LANGUAGE_VOICES } from "@/lib/languages";

interface UseElevenLabsTTSOptions {
  apiKey: string;
}

export function useElevenLabsTTS({ apiKey }: UseElevenLabsTTSOptions) {
  const audioContextRef = useRef<AudioContext | null>(null);
  const audioQueueRef = useRef<AudioBuffer[]>([]);
  const isPlayingRef = useRef(false);

  const speak = useCallback(async (text: string, language: string): Promise<void> => {
    const voiceId = LANGUAGE_VOICES[language] || LANGUAGE_VOICES.en;
    
    // Initialize AudioContext
    if (!audioContextRef.current) {
      audioContextRef.current = new AudioContext({ sampleRate: 24000 });
    }
    const ctx = audioContextRef.current;

    return new Promise((resolve, reject) => {
      // Connect to ElevenLabs TTS WebSocket
      const wsUrl = new URL(`wss://api.elevenlabs.io/v1/text-to-speech/${voiceId}/stream-input`);
      wsUrl.searchParams.set("model_id", "eleven_flash_v2_5");
      wsUrl.searchParams.set("output_format", "pcm_24000");
      wsUrl.searchParams.set("optimize_streaming_latency", "4");

      const ws = new WebSocket(wsUrl.toString());
      const audioChunks: Float32Array[] = [];

      ws.onopen = () => {
        // Initialize with space
        ws.send(JSON.stringify({ text: " " }));
        
        // Send full text and flush
        ws.send(JSON.stringify({ 
          text: text,
          flush: true,
          xi_api_key: apiKey,
        }));
      };

      ws.onmessage = async (event) => {
        const data = JSON.parse(event.data);
        
        if (data.audio) {
          // Decode base64 PCM audio
          const binaryString = atob(data.audio);
          const bytes = new Uint8Array(binaryString.length);
          for (let i = 0; i < binaryString.length; i++) {
            bytes[i] = binaryString.charCodeAt(i);
          }
          
          // Convert to Float32 for Web Audio
          const int16 = new Int16Array(bytes.buffer);
          const float32 = new Float32Array(int16.length);
          for (let i = 0; i < int16.length; i++) {
            float32[i] = int16[i] / 32768;
          }
          
          audioChunks.push(float32);
        }
        
        if (data.isFinal) {
          ws.close();
          
          // Concatenate all chunks
          const totalLength = audioChunks.reduce((sum, chunk) => sum + chunk.length, 0);
          const fullAudio = new Float32Array(totalLength);
          let offset = 0;
          for (const chunk of audioChunks) {
            fullAudio.set(chunk, offset);
            offset += chunk.length;
          }
          
          // Create AudioBuffer and play
          const audioBuffer = ctx.createBuffer(1, fullAudio.length, 24000);
          audioBuffer.getChannelData(0).set(fullAudio);
          
          const source = ctx.createBufferSource();
          source.buffer = audioBuffer;
          source.connect(ctx.destination);
          source.onended = () => resolve();
          source.start();
        }
      };

      ws.onerror = (error) => reject(error);
    });
  }, [apiKey]);

  return { speak };
}
```

### Phase 3: Token Endpoint (for browser WebSocket auth)

```typescript
export async function POST() {
  const response = await fetch("https://api.elevenlabs.io/v1/tokens", {
    method: "POST",
    headers: {
      "xi-api-key": process.env.ELEVENLABS_API_KEY!,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      // Token valid for 1 hour
      ttl_seconds: 3600,
    }),
  });

  const data = await response.json();
  return Response.json({ token: data.token });
}
```

**Why Groq Llama 3.1 8B Instant?**

```typescript
function CallUI({ ... }: VideoCallProps) {
  const [elevenLabsToken, setElevenLabsToken] = useState<string | null>(null);
  
  // Fetch ElevenLabs token on mount
  useEffect(() => {
    fetch("/api/elevenlabs-token", { method: "POST" })
      .then(res => res.json())
      .then(data => setElevenLabsToken(data.token));
  }, []);

  // ElevenLabs STT
  const { startListening, stopListening } = useElevenLabsSTT({
    apiKey: elevenLabsToken!,
    languageCode: "en", // Or detect from user
    onCommitted: (text) => {
      // Broadcast to all participants
      daily?.sendAppMessage({
        type: "transcription",
        text,
        user_id: visitorId,
        user_name: username,
      }, "*");
    },
  });

  // ElevenLabs TTS
  const { speak } = useElevenLabsTTS({ apiKey: elevenLabsToken! });

  // Start STT when joining (and not muted)
  useEffect(() => {
    if (!isJoining && !isMuted && elevenLabsToken) {
      navigator.mediaDevices.getUserMedia({ audio: true })
        .then(stream => startListening(stream));
    }
    return () => stopListening();
  }, [isJoining, isMuted, elevenLabsToken]);

  // Handle incoming transcriptions from other participants
  useDailyEvent("app-message", async (event) => {
    if (event?.data?.type !== "transcription") return;
    if (event.data.user_id === visitorId) return;

    const { text, user_name } = event.data;

    // Translate
    const translateRes = await fetch("/api/translate", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        text,
        fromLang: "auto",
        toLang: preferredLanguage,
      }),
    });
    const { translatedText } = await translateRes.json();

    // Update UI
    setCurrentTranslation(translatedText);
    setTranscripts(prev => [...prev.slice(-20), {
      id: crypto.randomUUID(),
      speaker: user_name,
      original: text,
      translated: translatedText,
      timestamp: new Date(),
    }]);

    // Play TTS via WebSocket
    await speak(translatedText, preferredLanguage);
    
    setTimeout(() => setCurrentTranslation(null), 3000);
  });

  // ... rest of component
}
```
